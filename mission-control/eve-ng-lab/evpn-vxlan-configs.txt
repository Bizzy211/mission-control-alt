================================================================================
  EVPN-VXLAN MULTI-TENANCY LAB — DEVICE CONFIGURATIONS
  Platform: Cisco Nexus 9000v (nxosv9k) on Eve-NG
================================================================================

EVE-NG REQUIREMENTS
-------------------
  Image:    nxosv9k-10.3.x or nxosv9k-9.3.x (QCOW2)
  RAM:      8 GB per Nexus 9000v (minimum)
  vCPU:     2 per device
  Total:    6 switches × 8 GB = 48 GB RAM + 4 hosts
  Hosts:    Use Linux (alpine/tiny) or VPCS for lightweight host emulation

  Tip: If 48 GB is too heavy, consider Arista vEOS (2-4 GB each) with
  equivalent EOS configs. The topology and overlay design are identical.

HOST ADDRESSING
---------------
  Host       Leaf     VLAN   IP Address        Default GW       Tenant
  --------   ------   ----   ---------------   --------------   --------
  HOST-A1    LEAF-1   10     192.168.10.10     192.168.10.1     Tenant-A
  HOST-B1    LEAF-2   20     192.168.20.10     192.168.20.1     Tenant-B
  HOST-A2    LEAF-3   10     192.168.10.20     192.168.10.1     Tenant-A
  HOST-B2    LEAF-4   20     192.168.20.20     192.168.20.1     Tenant-B

  Expected connectivity:
    HOST-A1 ↔ HOST-A2  ✓  (same Tenant-A, same VLAN 10, L2 stretched via VXLAN)
    HOST-B1 ↔ HOST-B2  ✓  (same Tenant-B, same VLAN 20, L2 stretched via VXLAN)
    HOST-A1 ↔ HOST-B1  ✗  (different VRFs — tenant isolation enforced)

ANYCAST GATEWAY MAC
-------------------
  All leafs share: 0000.2222.3333
  This enables seamless host mobility — the default gateway MAC is identical
  on every leaf, so hosts can move between leafs without ARP changes.

BGP DESIGN
----------
  AS:             65000 (iBGP)
  Address Family: L2VPN EVPN
  Spines:         Route Reflectors (all leafs are RR clients)
  Replication:    Ingress Replication (no multicast required)


================================================================================
  SPINE-1 CONFIGURATION
================================================================================

hostname SPINE-1

nv overlay evpn
feature ospf
feature bgp
feature nv overlay

router ospf UNDERLAY
  router-id 10.0.0.1

interface loopback0
  description Router-ID / BGP Peering
  ip address 10.0.0.1/32
  ip router ospf UNDERLAY area 0.0.0.0

interface Ethernet1/1
  description to LEAF-1 Eth1/1
  no switchport
  mtu 9216
  ip address 10.1.1.0/31
  ip router ospf UNDERLAY area 0.0.0.0
  no shutdown

interface Ethernet1/2
  description to LEAF-2 Eth1/1
  no switchport
  mtu 9216
  ip address 10.1.1.2/31
  ip router ospf UNDERLAY area 0.0.0.0
  no shutdown

interface Ethernet1/3
  description to LEAF-3 Eth1/1
  no switchport
  mtu 9216
  ip address 10.1.1.4/31
  ip router ospf UNDERLAY area 0.0.0.0
  no shutdown

interface Ethernet1/4
  description to LEAF-4 Eth1/1
  no switchport
  mtu 9216
  ip address 10.1.1.6/31
  ip router ospf UNDERLAY area 0.0.0.0
  no shutdown

router bgp 65000
  router-id 10.0.0.1
  address-family l2vpn evpn
    retain route-target all
  template peer LEAF
    remote-as 65000
    update-source loopback0
    address-family l2vpn evpn
      send-community extended
      route-reflector-client
  neighbor 10.0.0.11
    inherit peer LEAF
    description LEAF-1
  neighbor 10.0.0.12
    inherit peer LEAF
    description LEAF-2
  neighbor 10.0.0.13
    inherit peer LEAF
    description LEAF-3
  neighbor 10.0.0.14
    inherit peer LEAF
    description LEAF-4

  rd auto
  address-family ipv4 unicast
    route-target both auto
    route-target both auto evpn

  vni 50002
  rd auto
  address-family ipv4 unicast
    route-target both auto
    route-target both auto evpn

! ── Underlay (OSPF) ──────────────────────────────────────────────────
router ospf UNDERLAY
  router-id 10.0.0.11

interface loopback0
  description Router-ID / BGP Peering

  ip router ospf UNDERLAY area 0.0.0.0

  description VTEP Source
  ip address 10.0.1.11/32
  ip router ospf UNDERLAY area 0.0.0.0

interface Ethernet1/1

  no switchport
  mtu 9216
  ip address 10.1.1.1/31
  ip router ospf UNDERLAY area 0.0.0.0
  no shutdown

interface Ethernet1/2
  description to SPINE-2 Eth1/1
  no switchport
  mtu 9216
  ip address 10.1.2.1/31
  ip router ospf UNDERLAY area 0.0.0.0
  no shutdown

! ── Access Port ──────────────────────────────────────────────────────
interface Ethernet1/3
  description to HOST-A1
  switchport
  switchport mode access
  switchport access vlan 10
  no shutdown

! ── SVIs (Anycast Gateway) ───────────────────────────────────────────
interface Vlan10
  description Tenant-A Data Gateway
  no shutdown
  vrf member TENANT-A
  ip address 192.168.10.1/24
  fabric forwarding mode anycast-gateway

interface Vlan20
  description Tenant-B Data Gateway
  no shutdown
  vrf member TENANT-B
  ip address 192.168.20.1/24
  fabric forwarding mode anycast-gateway

interface Vlan901
  description Tenant-A L3VNI SVI
  no shutdown
  vrf member TENANT-A
  ip forward

interface Vlan902
  description Tenant-B L3VNI SVI
  no shutdown
  vrf member TENANT-B
  ip forward

! ── VXLAN (NVE) ──────────────────────────────────────────────────────
interface nve1
  no shutdown
  host-reachability protocol bgp
  source-interface loopback1
  member vni 10010
    ingress-replication protocol bgp
  member vni 10020
    ingress-replication protocol bgp
  member vni 50001 associate-vrf
  member vni 50002 associate-vrf

! ── BGP EVPN ─────────────────────────────────────────────────────────
router bgp 65000
  router-id 10.0.0.11
  address-family l2vpn evpn
  template peer SPINE
    remote-as 65000
    update-source loopback0
    address-family l2vpn evpn
      send-community extended
  neighbor 10.0.0.1
    inherit peer SPINE
    description SPINE-1
  neighbor 10.0.0.2
    inherit peer SPINE
    description SPINE-2
  vrf TENANT-A
    address-family ipv4 unicast
      advertise l2vpn evpn
  vrf TENANT-B
    address-family ipv4 unicast
      advertise l2vpn evpn

! ── EVPN VNI ─────────────────────────────────────────────────────────
evpn
  vni 10010 l2
    rd auto
    route-target import auto
    route-target export auto
  vni 10020 l2
    rd auto
    route-target import auto
    route-target export auto

STEP 1 — Verify Underlay (OSPF)
  show ip ospf neighbors
  show ip route ospf
  ping 10.0.0.1    ! from any leaf, verify spine loopback reachability
  ping 10.0.1.11   ! from any spine, verify VTEP loopback reachability

STEP 2 — Verify BGP EVPN Sessions
  show bgp l2vpn evpn summary
  ! All neighbors should be in "Established" state

STEP 3 — Verify NVE / VXLAN
  show nve peers
  show nve vni
  show vxlan interface

STEP 4 — Verify EVPN Routes
  show bgp l2vpn evpn
  show bgp l2vpn evpn vni-id 10010
  show bgp l2vpn evpn vni-id 10020
  show l2route evpn mac all
  show l2route evpn mac-ip all

STEP 5 — Verify Tenant Routing
  show ip route vrf TENANT-A
  show ip route vrf TENANT-B
  show ip arp vrf TENANT-A
  show ip arp vrf TENANT-B

STEP 6 — End-to-End Connectivity Tests
  ! From HOST-A1:
  ping 192.168.10.20    ← should succeed (HOST-A2, same tenant)
  ping 192.168.20.10    ← should FAIL   (HOST-B1, different tenant)

  ! From HOST-B1:
  ping 192.168.20.20    ← should succeed (HOST-B2, same tenant)
  ping 192.168.10.10    ← should FAIL   (HOST-A1, different tenant)

STEP 7 — EVPN Route Type Inspection
  show bgp l2vpn evpn route-type 2    ! MAC/IP routes (Type-2)
  show bgp l2vpn evpn route-type 3    ! Inclusive Multicast (Type-3)
  show bgp l2vpn evpn route-type 5    ! IP Prefix routes (Type-5)
